метод load() класса books:
void books::load()
{
	ifstream f("books.txt");

	if (f.is_open()) {
		while (!f.eof()) {
			book book;//объект класса книга
			//считывание из файла информации по книге
			f >> book.book_ID;
			f >> book.book_name;
			f >> book.book_author;
			f >> book.book_genre;
			f >> book.book_yaer;
			list_book.push_back(book);
		}
		f.close();
	}
}

Данный метод выполняет работу за O(n) и не требует оптимизации, для оптимального кода будет O(n).
