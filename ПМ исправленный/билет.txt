метод input() класса cruises:
void cruises :: input()
{
	ifstream file("cruise.txt");

	if (file.is_open()) 
	{
		while (!file.eof()) 
		{
			cruise cruise;
			file >> cruise.cruise_ID;
			file >> cruise.station_ID;
			file >> cruise.bus_ID;
			file >> cruise.departure_time;
			list_cruise.push_back(cruise);
		}
		file.close();
	}
}

Данный метод выполняет работу за O(n), для оптимального кода O(4n)
