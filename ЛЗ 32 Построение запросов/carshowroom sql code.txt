CREATE TABLE BUYERS (
    "BUYER CODE"  INTEGER NOT NULL,
    NAME          CHAR(30)
);

ALTER TABLE BUYERS ADD CONSTRAINT PK_BUYERS PRIMARY KEY ("BUYER CODE");

CREATE TABLE CARS (
    "CAR CODE"  INTEGER NOT NULL,
    MODEL       CHAR(20)
);

ALTER TABLE CARS ADD CONSTRAINT PK_CARS PRIMARY KEY ("CAR CODE");

CREATE TABLE MANAGERS (
    "MANAGER CODE"  INTEGER NOT NULL,
    NAME            CHAR(30) CHARACTER SET NONE
);

ALTER TABLE MANAGERS ADD CONSTRAINT PK_MANAGERS PRIMARY KEY ("MANAGER CODE");

CREATE TABLE SALES (
    "MANAGER CODE"   INTEGER,
    "CAR CODE"       INTEGER,
    "LICENSE PLATE"  CHAR(30),
    "BUYER CODE"     INTEGER,
    "DATE"           DATE,
    PRICE            INTEGER
);
